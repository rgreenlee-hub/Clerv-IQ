<script>
function calculateROI() {
  const timeSpent = parseFloat(document.getElementById("timeSpent").value) || 0;
  const clientValue = parseFloat(document.getElementById("clientValue").value) || 0;
  const callVolume = parseFloat(document.getElementById("callVolume").value) || 0;
  const conversionRate = (parseFloat(document.getElementById("conversionRate").value) || 0) / 100;

  // Simple model
  const potentialClients = callVolume * conversionRate;
  const revenueIncrease = potentialClients * clientValue;
  const timeSavingsValue = timeSpent * 4 * 50; // assume $50/hr equivalent efficiency

  const totalMonthlySavings = revenueIncrease + timeSavingsValue;
  const annualSavings = totalMonthlySavings * 12;
  const roi = annualSavings > 0 ? (annualSavings / 24000) * 100 : 0; // assume $2k/mo cost
  const payback = annualSavings > 0 ? (24000 / annualSavings) * 12 : 0;

  // Update UI
  document.getElementById("monthly-savings").textContent = "$" + totalMonthlySavings.toLocaleString();
  document.getElementById("annual-savings").textContent = "$" + annualSavings.toLocaleString();
  document.getElementById("roi-percentage").textContent = roi.toFixed(1) + "%";
  document.getElementById("payback-period").textContent = payback.toFixed(1) + " months";

  // Chart
  const ctx = document.getElementById('roiChart').getContext('2d');
  if (window.roiChartInstance) window.roiChartInstance.destroy();
  window.roiChartInstance = new Chart(ctx, {
    type: 'doughnut',
    data: {
      labels: ['Annual Savings', 'Annual Cost'],
      datasets: [{
        data: [annualSavings, 24000],
        backgroundColor: ['#00c8ff', '#333']
      }]
    },
    options: {
      responsive: true,
      plugins: {
        legend: { labels: { color: "#fff" } }
      }
    }
  });
}
</script>

document.getElementById("roiForm").addEventListener("submit", function(e) {

  const annualSavings = monthlySavings * 12;
  const roi = ((annualSavings - (currentCosts * 12)) / (currentCosts * 12)) * 100;
  const payback = (currentCosts * 12) / annualSavings * 12;

  // Update results
  document.getElementById("monthly-savings").textContent = `$${monthlySavings.toLocaleString()}`;
  document.getElementById("annual-savings").textContent = `$${annualSavings.toLocaleString()}`;

  document.getElementById("roiResults").classList.remove("hidden");

  // Chart.js visualization
  const ctx = document.getElementById("roiChart").getContext("2d");
  new Chart(ctx, {
    type: "doughnut",
    data: {
      labels: ["Annual Savings", "Annual Costs"],
      datasets: [{
        data: [annualSavings, currentCosts * 12],
        backgroundColor: ["#00c6ff", "#333"]
      }]
    }
  });
});
