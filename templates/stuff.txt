<script>
function calculateROI() {
  const currentCosts = parseFloat(document.getElementById("currentCosts").value) || 0;
  const timeSpent = parseFloat(document.getElementById("timeSpent").value) || 0;
  const hourlyRate = parseFloat(document.getElementById("hourlyRate").value) || 0;
  const callVolume = parseFloat(document.getElementById("callVolume").value) || 0;
  const conversionRate = parseFloat(document.getElementById("conversionRate").value) / 100 || 0;
  const efficiency = parseFloat(document.getElementById("efficiency").value) / 100 || 0;

  const monthlyCost = currentCosts;
  const timeSavings = timeSpent * efficiency * 4; // monthly hours saved
  const laborSavings = timeSavings * hourlyRate;
  const revenueIncrease = callVolume * conversionRate * 100; // est. added rev

  const totalMonthlySavings = laborSavings + revenueIncrease;
  const annualSavings = totalMonthlySavings * 12;
  const roi = ((annualSavings - monthlyCost * 12) / (monthlyCost * 12)) * 100;
  const payback = (monthlyCost * 12) / annualSavings * 12;

  // Update UI
  document.getElementById("monthly-savings").textContent = "$" + totalMonthlySavings.toLocaleString();
  document.getElementById("annual-savings").textContent = "$" + annualSavings.toLocaleString();
  document.getElementById("roi-percentage").textContent = roi.toFixed(1) + "%";
  document.getElementById("payback-period").textContent = payback.toFixed(1) + " months";

  // Chart
  const ctx = document.getElementById('roiChart').getContext('2d');
  if (window.roiChartInstance) window.roiChartInstance.destroy();
  window.roiChartInstance = new Chart(ctx, {
    type: 'doughnut',
    data: {
      labels: ['Annual Savings', 'Annual Cost'],
      datasets: [{
        data: [annualSavings, monthlyCost * 12],
        backgroundColor: ['#00c8ff', '#333']
      }]
    },
    options: {
      responsive: true,
      plugins: {
        legend: { labels: { color: "#fff" } }
      }
    }
  });
}
</script>
